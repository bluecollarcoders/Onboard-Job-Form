# ADR-001: Multi-Tenant Contact Platform Architecture

  ## Status
 **Proposed** | Date: 2026-02-26 | Decision Maker: Caleb Matteis

  ## Context
  Multi-tenant shared database
  ## Decision
 Multi-tenant with event-driven architecture

  ### Technical Implementation
  - Tenant isolation via tenantId foreign keys
  - Row-level security for data access
  - Event sourcing for audit trails
  - [Your detailed reasoning]

  ## Success Metrics
  - Query performance under load
  - Tenant isolation verification
  - Event processing latency

  ## Risks & Mitigation
  - Cross-tenant data leaks → Mitigation: RLS + integration tests
  - Event processing failures → Mitigation: Dead letter queues

  ## Consequences

// schema.prisma

// ==========================================
// 1. IDENTITY & ACCESS (The "Who")
// ==========================================

enum UserRole {
  OWNER    // Single point of accountability
  ADMIN    // Management & Billing
  MEMBER   // Standard functional access
}

model User {
  id              String    @id @default(cuid())
  email           String    @unique
  role            UserRole  @default(MEMBER)
  
  // Tenancy
  tenantId        String
  tenant          Tenant    @relation(fields: [tenantId], references: [id])
  
  // Security & Auth
  accounts        Account[] 
  sessions        Session[]
  isActive        Boolean   @default(true)
  lastLoginAt     DateTime?
  
  // Audit Trail
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt

  @@index([tenantId])
}

model Account {
  id                String  @id @default(cuid())
  userId            String
  type              String  // "credentials", "oauth", "saml"
  provider          String  // "google", "okta", "local"
  providerAccountId String  // Unique ID from the external IdP
  user              User    @relation(fields: [userId], references: [id])

  @@unique([provider, providerAccountId])
}

// ==========================================
// 2. TENANCY & GOVERNANCE (The "Bucket")
// ==========================================

enum PlanTier {
  FREE
  PRO
  ENTERPRISE
}

model Tenant {
  id              String    @id @default(cuid())
  name            String
  domain          String?   @unique // For SSO Discovery (e.g., "apple.com")
  plan            PlanTier  @default(FREE)
  
  // SSO / IdP Configuration
  ssoEnabled      Boolean   @default(false)
  ssoProviderUrl  String?   // The corporate login endpoint
  
  // Relationships
  users           User[]
  contacts        Contact[]
  
  // Meta
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
}

// ==========================================
// 3. BUSINESS DATA (The "What")
// ==========================================

model Contact {
  id              String    @id @default(cuid())
  name            String
  email           String
  
  // Multi-tenant Isolation
  tenantId        String
  tenant          Tenant    @relation(fields: [tenantId], references: [id])
  
  // Security & Audit
  createdBy       String?   // User who added the candidate
  deletedAt       DateTime? // Soft Delete (GDPR compliance)
  isPii           Boolean   @default(true)
  
  // Time-series Data
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt

  // PERFORMANCE INDEXES
  @@index([tenantId, createdAt]) // Fast dashboard sorting
  @@index([email])               // Fast global/tenant search
  @@unique([tenantId, email])    // Integrity: No dupes within one company
}

// ==========================================
// 4. SYSTEM RELIABILITY (The "How")
// ==========================================

model IntegrationOutbox {
  id          String   @id @default(cuid())
  eventType   String   // "CANDIDATE_CREATED", "USER_DEACTIVATED"
  payload     Json     
  processed   Boolean  @default(false)
  createdAt   DateTime @default(now())

  @@index([processed, createdAt]) // Efficient polling for workers
}
